\section{Email Structure}

COMMON EMAIL STRUCTURE:
\begin{enumerate}
\item Greeting
	\begin{enumerate}
	\item Salutation
	\item Pleasantry	
	\end{enumerate}
\item Body
	\begin{enumerate}
	\item Reference to the previous context
	\item Purpose of the email
	\item Giving details
	\item Call to action
	\end{enumerate}
\item Closing
	\begin{enumerate}
	\item Closing line
	\item Complimentary closing
	\item Signature
	\end{enumerate}
\end{enumerate}

MFU STRUCTURE:
\begin{enumerate}
\item Greeting
	\begin{enumerate}
	\item Salutation
	\end{enumerate}
\item Body
	\begin{enumerate}
	\item Reference to the previous meeting + a pleasantry
	\item Summary of the meeting
	\item List of action points: what should be done, who is responsible and by when
	\end{enumerate}
\item Closing
\end{enumerate}
